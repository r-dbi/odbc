579 inline std::string result::get<std::string>(short column, unsigned long row)
580 {
581  detail::bound_column& col = result_impl_get_bound_column(impl_, column, row);
582  char buffer[1024];
583  switch(col.ctype_)
584  {
585  case SQL_C_CHAR:
586  case SQL_C_GUID:
587  case SQL_C_BINARY:
588  if(col.blob_)
589  throw std::runtime_error("blob not implemented yet"); // #T implement load_blob(column);
590  return (col.pdata_ + row * col.clen_);
591 
592  case SQL_C_LONG:
593  if(std::snprintf(buffer, sizeof(buffer), "%ld", *(long*)(col.pdata_ + row * col.clen_)) != 1)
594  throw type_incompatible_error();
595  return buffer;
596 
597  case SQL_C_DOUBLE:
598  if(std::snprintf(buffer, sizeof(buffer), "%lf", *(double*)(col.pdata_ + row * col.clen_)) != 1)
599  throw type_incompatible_error();
600  return buffer;
601 
602  case SQL_C_DATE:
603  throw std::runtime_error("date not implemented yet"); // #T implement date conversion
604 
605  case SQL_C_TIMESTAMP:
606  throw std::runtime_error("timestamp not implemented yet"); // #T implement timestamp conversion
