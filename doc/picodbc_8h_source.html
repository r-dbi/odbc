574 template<>
575 inline std::string result::get<std::string>(short column, unsigned long row)
576 {
577  detail::bound_column& col = result_impl_get_bound_column(impl_, column, row);
578  char buffer[1024];
579  switch(col.ctype_)
580  {
581  case SQL_C_CHAR:
582  case SQL_C_GUID:
583  case SQL_C_BINARY:
584  if(col.blob_)
585  throw std::runtime_error("blob not implemented yet"); // #T implement load_blob(column);
586  return (col.pdata_ + row * col.clen_);
587 
588  case SQL_C_LONG:
589  if(std::snprintf(buffer, sizeof(buffer), "%ld", *(long*)(col.pdata_ + row * col.clen_)) != 1)
590  throw type_incompatible_error();
591  return buffer;
592 
593  case SQL_C_DOUBLE:
594  if(std::snprintf(buffer, sizeof(buffer), "%lf", *(double*)(col.pdata_ + row * col.clen_)) != 1)
595  throw type_incompatible_error();
596  return buffer;
597 
598  case SQL_C_DATE:
599  throw std::runtime_error("date not implemented yet"); // #T implement data conversion
600 
601  case SQL_C_TIMESTAMP:
602  throw std::runtime_error("timestamp not implemented yet"); // #T implement timestamp conversion
603  }
604  throw type_incompatible_error();
